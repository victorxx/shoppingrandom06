    function gerarStringUnica(tamanho = 120) {
      const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789!@#$%&*-_=+{}[]<>?';
      let resultado = '';
      for (let i = 0; i < tamanho; i++) {

      }
      return resultado;

    function carregarVideo(id) {

    // Cores para mudança automática no modo claro
    const coresFundo = [
      '#f8f6ff',  // branco lavanda claro (padrão)
      '#e0d7ff',  // lavanda suave
      '#f0e6ff',  // lilás clarinho
      '#efe9ff',  // lilás suave
      '#ddd6ff',  // roxo muito claro

    ];
    let corAtualIndex = 0;
    let intervaloCor;

    function iniciarMudancaDeCor() {
      intervaloCor = setInterval(() => {
        corAtualIndex = (corAtualIndex + 1) % coresFundo.length;
        document.body.style.backgroundColor = coresFundo[corAtualIndex];
      }, 4000);
    }

    function pararMudancaDeCor() {
      clearInterval(intervaloCor);
      document.body.style.backgroundColor = '#1e1a2e'; // fundo escuro fixo no modo dark
